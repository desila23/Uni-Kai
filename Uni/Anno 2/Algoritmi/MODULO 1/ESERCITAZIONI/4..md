UltimoZeroRic(A, i, f)

		then return -1
	
	m = [(i - f) / 2]
	
	if(L[m] == 0 && L[m+1] == 1) 
		then return m
	if(L[m] == 1) then 
		return UltimoZeroRic(A, k, i, m)

UltimoZero(A)
	n = lunghezza A
	if(A[n] == 0) 
		then return n
	if(A[1] == 1) then
		return -1
