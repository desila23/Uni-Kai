% ---------- Forma logica unica ----------
frase(mangia(S, O)) --> forma_mangia(S, O).
frase(mangia(S, O)) --> forma_si_nutre_di(S, O).
frase(mangia(S, O)) --> forma_cibo_e(S, O).


% ---------- Variante 1: "il cavallo mangia la biada"
forma_mangia(S, O) -->
    articolo, nome(S),
    [mangia],
    articolo, nome(O).


% ---------- Variante 2: "il cavallo si nutre di biada"
% (con o senza articolo sull'oggetto)
forma_si_nutre_di(S, O) -->
    articolo, nome(S),
    [si, nutre, di], nome(O).
forma_si_nutre_di(S, O) -->
    articolo, nome(S),
    [si, nutre, di], articolo, nome(O).


% ---------- Variante 3: "il cibo del cavallo è la biada"
forma_cibo_e(S, O) -->
    articolo, [cibo],
    prep_art_del, nome(S),
    [è],
    articolo, nome(O).


% ---------- Lessico: articoli, preposizioni articolate, nomi ----------
articolo     --> [il]  ; [lo]  ; [la]  ; [i]  ; [gli]  ; [le].
prep_art_del --> [del] ; [dello] ; [della] ; [dei] ; [degli] ; [delle].

nome(cavallo)    --> [cavallo].
nome(ragno)      --> [ragno].
nome(biada)      --> [biada].
nome(rosmarino)  --> [rosmarino].
nome(lupo)       --> [lupo].
nome(pecora)     --> [pecora].
